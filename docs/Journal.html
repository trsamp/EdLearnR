model = '
           auf =~ sa + rt + bp + imp
         aof =~ ao
         wef =~ we
         wef~ aof + auf
         aof ~ auf
         ao ~~ .01*ao
         we ~~ .01*we
         sa ~~ bp
     '
#summary(onefactor)
results = sem(model, sample.cov=au.cov, sample.nobs=605)
